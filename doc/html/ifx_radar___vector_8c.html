if ((input == NULL) || (output == NULL)){\
return IFX_ERROR_ARGUMENT_NULL;\
}\
if (output->length < input->length){\
return IFX_ERROR_DIMENSION_WRONG;\
}\
if ((factor > input->length)){\
return IFX_ERROR_ARGUMENT_INVALID;\
}\
uint32_t n1 = factor;\
uint32_t n2 = input->length - n1;\
if(input != output){\
memcpy(&output->data[n1], input->data, n2 * sizeof(ifx_Float_t));\
memcpy(output->data, &input->data[n2], n1 * sizeof(ifx_Float_t));\
}\
else if(input->length == 2*factor){\
for(uint32_t i = 0; i < factor; i++) {\
temp = input->data[i];\
input->data[i] = input->data[i + factor];\
input->data[i + factor] = temp;\
}\
}\
else {\
ifx_vector_create(input->length, temp_vec);\
memcpy(&temp_vec->data[n1], input->data, n2 * sizeof(ifx_Float_t));\
memcpy(temp_vec->data, &input->data[n2], n1 * sizeof(ifx_Float_t));\
memcpy(output->data, temp_vec->data, input->length * sizeof(ifx_Float_t));\
ifx_vector_destroy(temp_vec);\
}
