#define PREPROCESSED_FFT (   handle,
  input,
  preprocessing_result,
  mean,
  output,
  ret 
)
Value:
{\
if (handle->mean_removal_flag != 0){\
ret = ifx_math_mean(input, &mean);\
IFX_ERR_CHECK_RUNTIME(ret);\
}\
ret = ifx_math_substract_scalar(input, mean, preprocessing_result);\
IFX_ERR_CHECK_RUNTIME(ret);\
ret = ifx_math_multiply(&handle->fft_window, preprocessing_result);\
IFX_ERR_CHECK_RUNTIME(ret);\
ret = ifx_fft_run(handle->fft_handle, preprocessing_result, output);\
IFX_ERR_CHECK_RUNTIME(ret);\
}
